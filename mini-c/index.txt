int main(void) {
  int i
  auto n = (3 + 3) * 2

  if (n <= 0) {
    printf("Please enter a positive integer.\n")
    return -1
  }

  int t1 = 0
  int t2 = 1

  int nextTerm = t1 + t2

  printf("Fibonacci Series: %d, %d, ", t1, t2)

  for (i = 3; i <= n; ++i) {
    printf("%d, ", nextTerm)
    t1 = t2
    t2 = nextTerm
    nextTerm = t1 + t2
  }

  return 0;
}
